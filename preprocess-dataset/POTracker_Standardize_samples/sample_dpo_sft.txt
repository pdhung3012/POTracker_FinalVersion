Algorithm 1:

\documentclass{article}
\usepackage[linesnumbered,ruled,vlined]{algorithm2e}

\begin{document}

\begin{algorithm}[H]
\caption{Rule-based JSON to XML Transformation}
\KwIn{JSON object $J$}
\KwOut{XML document $X$}

\SetKwFunction{FTransform}{Transform}
\SetKwProg{Fn}{Function}{:}{}

\Fn{\FTransform{$J$, $parentTag$}}{
  \ForEach{key--value pair $(k, v)$ in $J$}{
    \uIf{$v$ is a primitive (string, number, boolean)}{
       Emit start-tag $<k>$\;
       Emit text $v$\;
       Emit end-tag $</k>$\;
    }
    \uElseIf{$v$ is \texttt{null}}{
       Emit empty tag $<k/>$\;
    }
    \uElseIf{$v$ is an object}{
       Emit start-tag $<k>$\;
       \FTransform{$v$, $k$}\;
       Emit end-tag $</k>$\;
    }
    \uElseIf{$v$ is an array}{
       Emit start-tag $<k>$\;
       \ForEach{element $e$ in $v$}{
          Emit start-tag $<item>$\;
          \FTransform{$e$, ``item''}\;
          Emit end-tag $</item>$\;
       }
       Emit end-tag $</k>$\;
    }
  }
}
\vspace{0.2cm}
\textbf{Main:}\;
\FTransform{Root JSON object $J$, ``root''}\;

\end{algorithm}

\end{document}


Algorithm 2:

\documentclass{article}
\usepackage[linesnumbered,ruled,vlined]{algorithm2e}

\begin{document}

\begin{algorithm}[H]
\caption{JSON to ODIN CIM IEC 61968-3 XML Transformation}
\KwIn{JSON outage report $J$}
\KwOut{XML document $X$ following ODIN schema}

\SetKwFunction{FTransform}{Transform}
\SetKwProg{Fn}{Function}{:}{}

\Fn{\FTransform{$J$, $parentTag$}}{

  \If{$parentTag$ is \texttt{root}}{
    Emit start-tag \texttt{<ns0:PubOutages>}\;
    Emit start-tag \texttt{<ns0:Outage>}\;
    Assign unique \texttt{mRID} from metadata\;
  }

  \ForEach{key--value pair $(k, v)$ in $J$}{
    \uIf{$k$ = ``name'' and $parentTag$ = ``county''}{
       Emit \texttt{<ns0:communityDescriptor>} $v$ \texttt{</ns0:communityDescriptor>}\;
    }
    \uElseIf{$k$ = ``cust\_s''}{
       Emit \texttt{<ns0:metersServed>} $v$ \texttt{</ns0:metersServed>}\;
    }
    \uElseIf{$k$ = ``cust\_a.val''}{
       Emit \texttt{<ns0:metersAffected>} $v$ \texttt{</ns0:metersAffected>}\;
    }
    \uElseIf{$k$ = ``areas'' (array)}{
       \ForEach{element $e$ in $v$}{
          Emit start-tag \texttt{<ns0:OutageArea>}\;
          Map $e.name$ → \texttt{<ns0:communityDescriptor>}\;
          Map $e.cust\_s$ → \texttt{<ns0:metersServed>}\;
          Map $e.cust\_a.val$ → \texttt{<ns0:metersAffected>}\;
          Emit fixed tag \texttt{<ns0:outageAreaKind>township</ns0:outageAreaKind>}\;
          Emit end-tag \texttt{</ns0:OutageArea>}\;
       }
    }
    \uElseIf{$k$ = ``county.cust\_s''}{
       Emit \texttt{<ns0:OutageArea>} with
       \texttt{<ns0:metersServed>} $v$ \texttt{</ns0:metersServed>}
       and \texttt{<ns0:outageAreaKind>county</ns0:outageAreaKind>}\;
    }
  }

  \If{$parentTag$ is \texttt{root}}{
    Emit utility identifiers under \texttt{<ns0:Names>}\;
    Emit end-tag \texttt{</ns0:Outage>}\;
    Emit end-tag \texttt{</ns0:PubOutages>}\;
  }

}

\vspace{0.2cm}
\textbf{Main:}\;
\FTransform{Root JSON object $J$, ``root''}\;

\end{algorithm}

\end{document}


Rule changed:

Rule 1 (Object): County JSON object → <ns0:Outage> with county-level tags.

Rule 2 (Array): areas[] JSON array → multiple <ns0:OutageArea>.

Rule 3 (Primitive): Numeric/string values placed as tag text.

Rule 4 (Nulls): Dropped if not required by CIM (e.g., etr is omitted since null).

Rule 5 (Standard mapping): JSON keys mapped to CIM tags instead of literal tag names (cust_s → <ns0:metersServed>).
